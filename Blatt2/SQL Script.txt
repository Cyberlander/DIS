VERTICAL PARTITIONING

CREATE TABLE ESTATE_AGENT ( 
    NAME VARCHAR(255) NOT NULL, 
    ADDRESS VARCHAR(255), 
    LOGIN VARCHAR(255) NOT NULL, 
    PASSWORD VARCHAR(255) NOT NULL, 
    PRIMARY KEY(NAME) );

CREATE TABLE ESTATE ( 
	ID INTEGER NOT NULL,
	CITY varchar(255),
	POSTAL_CODE INTEGER,
	STREET varchar(255),
	STREET_NUMBER varchar(255),
	SQUARE_AREA INTEGER,
	ESTATE_AGENT VARCHAR(255),
	PRIMARY KEY(ID),
	CONSTRAINT fk_estate_manager FOREIGN KEY(ESTATE_AGENT) REFERENCES ESTATE_AGENT(NAME)); 

CREATE TABLE APARTMENT (
	ID INTEGER NOT NULL,
	FLOOR INTEGER,
	RENT INTEGER,
	ROOMS INTEGER,
	BALCONY INTEGER,
	BUILT_IN_KITCHEN VARCHAR(255),
	PRIMARY KEY(ID),
	CONSTRAINT fk_apartment_estate FOREIGN KEY(ID) REFERENCES ESTATE(ID) );

CREATE TABLE HOUSE ( 
	ID INTEGER NOT NULL,
	FLOOR INTEGER,
	PRICE INTEGER,
	GARDEN VARCHAR(255),
	PRIMARY KEY(ID),
	CONSTRAINT fk_house_estate FOREIGN KEY(ID) REFERENCES ESTATE(ID) );

// GENERATED ALWAYS AS IDENTITY (START WITH 1 INCREMENT BY 1)
CREATE TABLE PERSON (
	ID INTEGER NOT NULL,
	FIRST_NAME VARCHAR(255),
	NAME VARCHAR(255),
	ADDRESS VARCHAR(255),
	PRIMARY KEY(ID) );

CREATE TABLE TENANCY_CONTRACT (
	CONTRACT_NO INTEGER NOT NULL,
	START_DATE VARCHAR(255),
	DURATION VARCHAR(255),
	ADDITIONAL_COSTS VARCHAR(255),
	PRIMARY KEY(CONTRACT_NO),
	CONSTRAINT fk_contract_tenancy FOREIGN KEY(CONTRACT_NO)
	REFERENCES CONTRACT(CONTRACT_NO) );

CREATE TABLE PURCHASE_CONTRACT (
	CONTRACT_NO INTEGER NOT NULL,
	NO_OF_INSTALLMENTS INTEGER,
	INTREST_RATE VARCHAR(255),
	PRIMARY KEY(CONTRACT_NO),
	CONSTRAINT fk_purchase_contract FOREIGN KEY(CONTRACT_NO)
	REFERENCES CONTRACT(CONTRACT_NO) );

CREATE TABLE CONTRACT (
	CONTRACT_NO INTEGER NOT NULL,
	DATE_ VARCHAR(255),
	PLACE VARCHAR(255),
	PRIMARY KEY(CONTRACT_NO) );
